## 先读这里：你已经知道的东西

你有 Python 和 C 基础，这是非常好的起点。让我告诉你它们和 ArkTS 的关系：

| 概念 | Python | C | ArkTS（你要学的） |
|------|--------|---|-----------------|
| 变量类型 | 动态（运行时确定） | 静态（编译时确定） | 静态（和 C 类似，但更友好） |
| 函数 | `def func():` | `void func(){}` | `function func(){}` |
| 类 | `class Foo:` | 结构体+指针 | `class Foo {}` |
| 数组 | `[1,2,3]` | `int arr[3]` | `let arr: number[] = [1,2,3]` |
| 条件 | `if x > 0:` | `if(x > 0)` | `if (x > 0)` |
| 循环 | `for i in range(10):` | `for(int i=0;i<10;i++)` | `for (let i=0; i<10; i++)` |
| 注释 | `# 注释` | `// 注释` | `// 注释`（和 C 一样） |
| 打印 | `print("hello")` | `printf("hello")` | `console.log("hello")` |

**结论**：ArkTS 的语法比 C 更简洁，类型系统和 C 类似，但不需要管内存。你上手的速度会比想象中快。

### 本周目标（用 Python 理解对比学习）

**目标1：变量声明和类型**
```typescript
// Python 写法（你熟悉的）
name = "张三"
age = 25
height = 1.75
is_member = True

// ArkTS 写法（你要学的）
let name: string = "张三"     // let = 可变量
let age: number = 25
let height: number = 1.75
let isMember: boolean = true  // 注意：true 小写

const PI = 3.14159            // const = 常量（相当于 Python 的全大写约定）
```

**目标2：函数**
```typescript
// Python
def calculate_bmi(weight_kg: float, height_m: float) -> float:
    return weight_kg / (height_m ** 2)

// ArkTS（参数类型写在参数名后面，返回类型写在括号后面）
function calculateBmi(weightKg: number, heightM: number): number {
    return weightKg / (heightM * heightM)
}

// 调用方式一样：
let bmi = calculateBmi(72.5, 1.75)
console.log(`BMI: ${bmi.toFixed(1)}`)   // toFixed(1) 保留1位小数
```

**目标3：类（Class）**
```typescript
// Python
class WeightRecord:
    def __init__(self, date: str, weight_kg: float):
        self.date = date
        self.weight_kg = weight_kg
    
    def get_bmi(self, height_m: float) -> float:
        return self.weight_kg / (height_m ** 2)

// ArkTS
class WeightRecord {
    date: string         // 属性声明（Python 在 __init__ 里声明）
    weightKg: number
    
    constructor(date: string, weightKg: number) {   // constructor = Python 的 __init__
        this.date = date          // this = Python 的 self
        this.weightKg = weightKg
    }
    
    getBmi(heightM: number): number {
        return this.weightKg / (heightM * heightM)
    }
}

// 使用
let record = new WeightRecord("2026-05-17", 72.5)
console.log(record.getBmi(1.75))
```

**目标4：数组操作**
```typescript
// Python                          // ArkTS
weights = [72.5, 72.1, 71.8]     let weights: number[] = [72.5, 72.1, 71.8]
weights.append(71.5)              weights.push(71.5)         // 追加
weights[0]                        weights[0]                 // 按索引
len(weights)                      weights.length             // 长度

// 遍历（Python 的 for item in list）
for weight in weights:            for (let weight of weights) {
    print(weight)                     console.log(weight)
                                  }

// 过滤（Python 的列表推导式）
# [w for w in weights if w > 72]
weights.filter(w => w > 72)      // 箭头函数 => 相当于 Python 的 lambda

// 映射
# [w * 2.205 for w in weights]   （kg 转 lb）
weights.map(w => w * 2.205)
```

**目标5：接口（Interface）**
```typescript
// ArkTS 特有概念，Python 里没有直接对应，但类似 TypedDict 或 dataclass

// Python (dataclass)             // ArkTS (interface)
@dataclass                        interface WeightRecord {
class WeightRecord:                   id: number
    id: int                           date: string
    date: str                         weightKg: number
    weight_kg: float                  bmi?: number    // ? = 可选
    bmi: float = None             }

// 用法：定义函数接受的数据形状
function saveWeight(record: WeightRecord): void {
    // record 一定有 id, date, weightKg，可能有 bmi
}
```

## 第二阶段（第 3-4 周）：ArkUI 界面开发基础

### 核心概念：声明式 UI

```typescript
// ArkUI 的哲学：描述界面"长什么样"，不是"怎么画"

// 就像你告诉设计师"这里要一个蓝色大字"，
// 而不是自己一像素一像素地画

@Component
struct WeightDisplay {
    @State weight: number = 72.5    // @State 变量变化时，界面自动刷新
    
    build() {
        Column() {                   // Column = 纵向排列（从上到下）
            Text('今日体重')
                .fontSize(14)
                .fontColor('#888888')
            
            Text(`${this.weight} kg`)
                .fontSize(32)
                .fontWeight(FontWeight.Bold)
                .fontColor('#378ADD')
            
            Button('更新体重')
                .onClick(() => {
                    this.weight = 71.8    // 点击后 @State 变量改变，界面自动更新
                })
        }
        .padding(16)
    }
}
```

### 常用布局组件

```typescript
Column() { /* 纵向排列，像 Python 的竖着叠 */ }
Row() { /* 横向排列，像 Python 的并排放 */ }
Stack() { /* 层叠，像 PS 的图层 */ }
List() { /* 列表（支持懒加载，千条数据也流畅） */ }
Grid() { /* 网格布局 */ }
Scroll() { /* 可滚动容器 */ }
```

### 状态管理（ArkTS 最重要的概念）

```typescript
// @State：组件自己的状态
// 类比：Python 类的实例变量，但变化时界面会自动刷新

@State private count: number = 0

// @Prop：从父组件传入（只能读，不能改）
// 类比：Python 函数的参数

@Prop targetWeight: number = 70

// @Link：双向绑定（父子都能改）
// 类比：Python 传入可变对象（list/dict）

@Link sharedWeight: number

// @StorageLink：全局状态（整个 APP 都能访问）
// 类比：Python 的全局变量（但更安全）

@StorageLink('currentUser') user: UserInfo = new UserInfo()
```

### 数据库基本操作

```typescript
// 在 ArkTS 里操作数据库（和 Python 的 sqlite3 模块很像）

import relationalStore from '@ohos.data.relationalStore'

// 插入一条体重记录
async function insertWeight(weight: number, date: string): Promise<void> {
    const store = await getDatabase()    // 获取数据库连接
    
    const valueBucket: relationalStore.ValuesBucket = {
        'weight_kg': weight,
        'record_date': date,
        'bmi': weight / (1.75 * 1.75)
    }
    
    await store.insert('weight_records', valueBucket)
    console.log('体重记录保存成功')
}

// 查询最近7天的体重
async function queryWeeklyWeights(): Promise<WeightRecord[]> {
    const store = await getDatabase()
    
    const predicates = new relationalStore.RdbPredicates('weight_records')
    predicates.greaterThanOrEqualTo('record_date', getDateDaysAgo(7))
    predicates.orderByAsc('record_date')
    
    const resultSet = await store.query(predicates, ['id', 'record_date', 'weight_kg'])
    
    const records: WeightRecord[] = []
    while (resultSet.goToNextRow()) {
        records.push({
            id: resultSet.getLong(0),
            date: resultSet.getString(1),
            weightKg: resultSet.getDouble(2)
        })
    }
    resultSet.close()
    return records
}
```

## 学习过程中遇到问题怎么办？

1. **优先查官方文档**：https://developer.huawei.com/consumer/cn/doc/harmonyos-guides/arkts-overview
2. **在文档里搜索组件名**：比如遇到 `List` 组件不会用，直接搜 "List ArkUI"
3. **报错看日志**：在 DevEco Studio 底部的 "Log" 面板查看错误，红色的就是错误
4. **常见错误处理**：
   - `Cannot read property of null`：你访问了空对象的属性，检查变量是否初始化
   - `Type 'string' is not assignable to type 'number'`：类型不对，检查变量类型
   - `Unresolved reference`：变量名拼错了，或者忘记 import
